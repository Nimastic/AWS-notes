%% ── Exam-level tag commands ──────────────────────────────────
% Inline tags — \texorpdfstring makes them safe inside section headings
% (hyperref would hang on bare \tikz in moving arguments)
\newcommand{\saatagrender}{%
  \tikz[baseline=-0.5ex]\node[fill=saacolor, rounded corners=2pt,
    inner sep=2pt, text=white, font=\bfseries\tiny]{SAA};}
\newcommand{\saptagrender}{%
  \tikz[baseline=-0.5ex]\node[fill=sapcolor, rounded corners=2pt,
    inner sep=2pt, text=white, font=\bfseries\tiny]{SAP};}
\newcommand{\bothtagrender}{%
  \tikz[baseline=-0.5ex]\node[fill=sharedcolor, rounded corners=2pt,
    inner sep=2pt, text=white, font=\bfseries\tiny]{SAA+SAP};}

\newcommand{\saatag}{\texorpdfstring{\saatagrender}{[SAA]}}
\newcommand{\saptag}{\texorpdfstring{\saptagrender}{[SAP]}}
\newcommand{\bothtag}{\texorpdfstring{\bothtagrender}{[SAA+SAP]}}
