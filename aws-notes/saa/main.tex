\section{SAA-C03 at a Glance}

\begin{keyfacts}
\begin{itemize}
  \item \textbf{Questions:} 65 scored + 15 unscored (not labelled) = 80 total
  \item \textbf{Duration:} 130 minutes
  \item \textbf{Passing score:} 720 / 1000
  \item \textbf{Format:} Multiple choice (1 correct) + Multiple response (2+ correct)
  \item \textbf{Validity:} 3 years
\end{itemize}
\end{keyfacts}
