\section{SAP-C02 at a Glance}

\begin{keyfacts}
\begin{itemize}
  \item \textbf{Questions:} 65 scored + 10 unscored = 75 total
  \item \textbf{Duration:} 180 minutes
  \item \textbf{Passing score:} 750 / 1000
  \item \textbf{Format:} Multiple choice + Multiple response (often 3-of-5 or 2-of-5)
  \item \textbf{Prerequisite (recommended):} SAA-C03 or equivalent experience
\end{itemize}
\end{keyfacts}

\subsection{Domain Weightings}

\begin{comparebox}[title={SAP-C02 content domains}]
\begin{tabular}{@{}clc@{}}
\toprule
\textbf{Domain} & \textbf{Name} & \textbf{Weight} \\
\midrule
1 & Design Solutions for Organizational Complexity    & 26\% \\
2 & Design for New Solutions                          & 29\% \\
3 & Continuous Improvement for Existing Solutions     & 25\% \\
4 & Accelerate Workload Migration and Modernization   & 20\% \\
\bottomrule
\end{tabular}
\end{comparebox}

\textbf{How SAP differs from SAA:}
\begin{itemize}
  \item Scenarios are longer and involve \emph{trade-offs} between multiple valid options
  \item Heavy focus on \textbf{organisational complexity}: multi-account, hybrid, enterprise
  \item Migration and modernisation at scale
  \item Cost and operational \emph{governance}, not just individual architecture
\end{itemize}
