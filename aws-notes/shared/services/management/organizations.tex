\begin{keyfacts}
\begin{itemize}
  \item \textbf{Management Account} (formerly Master) --- root of the org; creates/invites accounts; \textbf{NOT} affected by SCPs
  \item \textbf{Organizational Units (OUs)} --- logical groupings; can be nested; SCPs apply to all accounts in OU
  \item \textbf{Service Control Policies (SCPs)} --- set \emph{maximum} permission boundaries; do NOT grant permissions; must be combined with IAM policies
  \item \textbf{Consolidated Billing} --- aggregate usage for volume discounts; single payment method; RI/Savings Plan sharing across accounts
\end{itemize}
\end{keyfacts}

\begin{gotcha}
\textbf{SCPs do NOT grant permissions} --- they only restrict the maximum permissions available. If an SCP has an explicit Deny, not even the account root user can perform that action. If an SCP has Allow, the IAM policy in the account must also Allow the action before it succeeds.
\end{gotcha}

\subsubsection{SCP Example: Restrict to Specific Regions}
\begin{verbatim}
{
  "Effect": "Deny",
  "NotAction": ["iam:*", "support:*", "sts:*"],
  "Resource": "*",
  "Condition": {
    "StringNotEquals": {
      "aws:RequestedRegion": ["us-east-1", "ap-southeast-1"]
    }
  }
}
\end{verbatim}
\textit{\small NotAction is used so IAM/STS (global services) still work regardless of region.}

\subsection{AWS Control Tower}

\begin{keyfacts}
\begin{itemize}
  \item \textbf{Landing Zone} --- a well-architected, multi-account baseline environment set up automatically
  \item \textbf{Guardrails} --- pre-configured governance rules:
    \begin{itemize}
      \item \emph{Preventive} guardrails = SCPs (e.g.\ disallow root user access keys)
      \item \emph{Detective} guardrails = AWS Config rules (e.g.\ detect public S3 buckets)
    \end{itemize}
  \item \textbf{Account Factory} --- automates new account provisioning with approved baseline configurations; built on Service Catalog
  \item \textbf{Log Archive Account} --- centralised CloudTrail and Config logs for all accounts
  \item \textbf{Audit Account} --- read/write access to all accounts for security and compliance teams
\end{itemize}
\end{keyfacts}

\begin{examtip}
\textbf{Organizations vs Control Tower:} AWS Organizations is the underlying service; Control Tower is an opinionated, automated layer on top. If you need to set up a multi-account environment quickly with best practices baked in, Control Tower is the answer. If you need fine-grained custom control, use Organizations + SCPs directly.
\end{examtip}

\subsection{Multi-Account Logging Strategy}

\begin{keyfacts}
\begin{itemize}
  \item Centralise \textbf{CloudTrail} logs to a dedicated \emph{Log Archive} account S3 bucket; enable org-level trail from management account
  \item Use \textbf{S3 bucket policies} to allow writes (from member accounts) but \emph{deny deletes} --- prevents tampering
  \item Enable \textbf{AWS Config} with an aggregator to view compliance across all accounts from a single pane
  \item Use \textbf{Amazon EventBridge} for cross-account event routing and notifications
  \item \textbf{AWS Security Hub} --- aggregate findings from GuardDuty, Inspector, Macie, Config across all accounts; delegate admin to security account
\end{itemize}
\end{keyfacts}

\subsection{AWS Resource Access Manager (RAM)}

\begin{keyfacts}
\begin{itemize}
  \item Share AWS resources across accounts \emph{without VPC peering}
  \item Shareable resources include: VPC subnets, Transit Gateway, Route 53 Resolver rules, License Manager configurations, Capacity Reservations
  \item \textbf{Shared VPC pattern:} central networking account owns the VPC and subnets; spoke accounts deploy resources into those subnets via RAM
  \item Does not copy resources --- the resource stays in the owner account; RAM grants permissions
\end{itemize}
\end{keyfacts}

\subsubsection{Tagging Strategy for Cost Allocation}
\begin{itemize}
  \item Tag all resources: \texttt{CostCenter}, \texttt{Project}, \texttt{Environment} (prod/dev/staging), \texttt{Owner}
  \item \textbf{Activate cost allocation tags} in Billing Console to see them in Cost Explorer
  \item Use \textbf{AWS Config rules} or \textbf{SCPs} to enforce mandatory tagging on resource creation
  \item \textbf{AWS Budgets with actions} --- automatically apply an SCP or stop EC2 instances when budget thresholds are breached
\end{itemize}
