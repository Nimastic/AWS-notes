\begin{comparebox}[title={7Rs Quick Reference}]
\begin{tabular}{@{}p{2.5cm}p{2.5cm}p{6cm}@{}}
\toprule
\textbf{Strategy} & \textbf{Also Known As} & \textbf{When to Use} \\
\midrule
Retire        & Remove             & App no longer needed; decommission \\
Retain        & Keep               & Too risky, not worth migrating; revisit later \\
Rehost        & Lift \& Shift      & Fast migration; no code changes; use MGN \\
Relocate      & Lift \& Shift      & VMware on-prem to VMware Cloud on AWS \\
Replatform    & Lift \& Reshape    & Minor optimisations (MySQL on EC2 $\to$ RDS); minimal code changes \\
Repurchase    & Drop \& Shop       & Replace with SaaS (on-prem CRM $\to$ Salesforce) \\
Refactor      & Re-architect       & Max cloud benefit; rethink as microservices/serverless; highest cost \\
\bottomrule
\end{tabular}
\end{comparebox}

\begin{examtip}
\textbf{Rehost = fastest, lowest effort.} \textbf{Refactor = most expensive, best long-term benefit.} Do not overlook \textbf{Retire} and \textbf{Retain} --- they are valid answers when the question mentions ``budget constraints'' or ``app will be sunsetted within 6 months.''
\end{examtip}

\subsection{Snow Family}

\begin{comparebox}[title={Snow Family devices}]
\begin{tabular}{@{}p{3.5cm}p{5.5cm}p{3cm}@{}}
\toprule
\textbf{Device} & \textbf{Description} & \textbf{Capacity} \\
\midrule
Snowcone        & Smallest; portable; edge computing + data transfer; rugged        & 8--14 TB \\
Snowball Edge   & Storage-Optimised (80 TB) or Compute-Optimised (ML at edge)       & 80 TB usable \\
Snowmobile      & Semi-truck for exabyte-scale migrations; AWS manages logistics     & Up to 100 PB \\
\bottomrule
\end{tabular}
\end{comparebox}

\begin{examtip}
\textbf{Rule of thumb:} if transferring the data over the network would take \emph{more than 1 week}, use Snow Family. For petabytes, use Snowmobile. For terabytes with edge compute needs, use Snowball Edge. For ongoing incremental sync (not bulk), use DataSync.
\end{examtip}

\subsection{Database Migration Patterns}

\begin{keyfacts}
\begin{itemize}
  \item \textbf{Homogeneous migration} (e.g.\ MySQL $\to$ RDS MySQL, Oracle $\to$ Oracle) --- use \awsservice{DMS} directly; no schema conversion needed
  \item \textbf{Heterogeneous migration} (e.g.\ Oracle $\to$ Aurora PostgreSQL) --- use \awsservice{SCT} first to convert schema, then \awsservice{DMS} for data
  \item \textbf{CDC (Change Data Capture)} --- DMS supports ongoing replication after initial load; minimises cutover downtime
  \item \textbf{Large database offline load} --- use DMS + Snowball Edge for initial bulk load, then switch to CDC over the network for the delta
\end{itemize}
\end{keyfacts}

\begin{saadepth}
SAA: know Snow Family selection (Snowcone vs Snowball vs Snowmobile by size), DMS for database migrations, DataSync for file/NFS/SMB transfers.
\end{saadepth}

\begin{sapdepth}
SAP: know the full 7Rs and when each applies. Understand wave planning (group workloads by dependencies into migration waves). Key modernisation patterns:
\begin{itemize}
  \item Oracle/SQL Server $\to$ Aurora (reduce licensing cost; Babelfish for SQL Server compat with Aurora PostgreSQL)
  \item MySQL/PostgreSQL $\to$ Aurora (up to 5$\times$ MySQL, 3$\times$ PostgreSQL performance)
  \item Self-managed Elasticsearch $\to$ Amazon OpenSearch Service
  \item Data warehouse $\to$ Amazon Redshift (petabyte-scale analytics, columnar storage)
  \item Strangler Fig pattern: gradually replace monolith functionality with microservices routing traffic via API Gateway or EventBridge
\end{itemize}
\end{sapdepth}
