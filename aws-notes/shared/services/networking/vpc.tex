\subsection{Core Components}

\begin{keyfacts}
\begin{itemize}
  \item \textbf{CIDR block} — IPv4 /16 to /28; IPv6 /56 (optional)
  \item \textbf{Subnets} — within one AZ; public (route to IGW) vs private
  \item \textbf{Internet Gateway (IGW)} — allows public subnet $\leftrightarrow$ internet
  \item \textbf{Route Tables} — one per subnet (or shared); most specific route wins
  \item \textbf{NAT Gateway} — lets private subnets reach internet (outbound only).
        Managed, \cost{charged per GB + per hour. AZ-specific; deploy one per AZ for HA.}
  \item \textbf{NACL} — subnet-level, stateless firewall; rules evaluated in order
  \item \textbf{Security Group} — instance-level, stateful; no explicit deny (only allow)
\end{itemize}
\end{keyfacts}

\begin{gotcha}
VPC Peering is \textbf{not transitive}: if VPC A peers with B, and B peers with C,
A cannot reach C through B. Use \awsservice{Transit Gateway} for hub-and-spoke
connectivity across many VPCs.
\end{gotcha}
